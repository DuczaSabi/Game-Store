-- FUNCTION: public.sortgenre(integer, integer, integer)

-- DROP FUNCTION IF EXISTS public.sortgenre(integer, integer, integer);

CREATE OR REPLACE FUNCTION public.sortgenre(
	genreid integer,
	pagenumber integer,
	pagelimit integer)
    RETURNS SETOF "Game" 
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
    ROWS 1000

AS $BODY$

BEGIN
	IF genreID = 0 THEN
	RETURN QUERY
        SELECT *
        FROM "Game"
        ORDER BY "Game"."Id"
		LIMIT(pageLimit)
		OFFSET((pageNumber -1) * pageLimit);
	ELSE
    RETURN QUERY
        SELECT *
        FROM "Game"
        WHERE "Game"."Genre" = genreID
		ORDER BY "Game"."Id"
		LIMIT(pageLimit)
		OFFSET((pageNumber -1) * pageLimit);
	END IF;
END 
$BODY$;

ALTER FUNCTION public.sortgenre(integer, integer, integer)
    OWNER TO postgres;
