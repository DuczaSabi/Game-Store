-- FUNCTION: public.updategame(character varying, character varying, character varying, character varying, character varying, integer)

-- DROP FUNCTION IF EXISTS public.updategame();

CREATE OR REPLACE FUNCTION public.updategame(selectedGame CHARACTER VARYING, title CHARACTER VARYING, image CHARACTER VARYING, genre CHARACTER VARYING, refurl CHARACTER VARYING, price INTEGER)
RETURNS SETOF "Game"
LANGUAGE 'plpgsql'
AS $$
		
BEGIN
	RETURN QUERY
		UPDATE "Game"
		SET "Title" = title, "Image" = image, "Genre" = genre, "Link" = refurl, "Price" = price
		WHERE "Title" LIKE selectedGame;
END $$ ;
