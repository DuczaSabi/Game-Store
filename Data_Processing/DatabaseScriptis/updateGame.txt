-- FUNCTION: public.updategame(integer, character varying, character varying, character varying, character varying, integer)

-- DROP FUNCTION IF EXISTS public.updategame(integer, character varying, character varying, character varying, character varying, integer);

CREATE OR REPLACE FUNCTION public.updategame(
	gameid integer,
	title character varying,
	image character varying,
	genre character varying,
	refurl character varying,
	price integer)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
		
BEGIN
		UPDATE "Game"
		SET "Title" = title, "Image" = image, "Genre" = genre, "Link" = refurl, "Price" = price
		WHERE "Id" = gameId;
END 
$BODY$;

ALTER FUNCTION public.updategame(integer, character varying, character varying, character varying, character varying, integer)
    OWNER TO postgres;
